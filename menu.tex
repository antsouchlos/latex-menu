\newcommand{\frontcoverdecorations}[1]{%
  \begin{tikzpicture}[remember picture, overlay]
    \node[anchor=north west, xshift=0.5cm, yshift=-0.5cm] (NW) at (current page.north west){%
      \pgfornament[width=2cm,color=ornamentcolor]{#1}};
    \node[anchor=north east, xshift=-0.5cm, yshift=-0.5cm] (NE) at (current page.north east){%
      \pgfornament[width=2cm,color=ornamentcolor,symmetry=v]{#1}};
    \node[anchor=south west, xshift=0.5cm, yshift=0.5cm] (SW) at (current page.south west){%
      \pgfornament[width=2cm,color=ornamentcolor,symmetry=h]{#1}};
    \node[anchor=south east, xshift=-0.5cm, yshift=0.5cm] (SE) at (current page.south east){%
      \pgfornament[width=2cm,color=ornamentcolor,symmetry=c]{#1}};

  \draw[ornamentcolor, line width=.5mm, double distance=0.5mm, double=bgcolor]  (NW.north west) -- (NE.north east) -- (SE.south east) -- (SW.south west) -- cycle;

  \end{tikzpicture}%
}
